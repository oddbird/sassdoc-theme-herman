  'root': 18px,

  'base-rem': 'root' ('convert-units': 'rem'),
  'small-rem': 16px ('convert-units': 'rem'),
  'xsmall-rem': 'base-rem' ('times': 0.75),

  'large': calc(%s + %s) ('base-rem', 1vw),
  'medium': calc(%s + %s) ('base-rem', 0.5vw),
  'base': calc(%s + %s) ('small-rem', 0.5vw),
  'small': calc(%s + %s) ('small-rem', 0.25vw),
  'xsmall': calc(%s + %s) ('xsmall-rem', 0.25vw),
  'code': 'xsmall',

  'double-gutter': 'gutter' ('times': 2),
  'spacer': 'gutter' ('times': 3),
  'double-spacer': 'spacer' ('times': 2),
  'shim': 'gutter' ('times': 0.5),
