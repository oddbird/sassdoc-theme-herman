Code Blocks

scss 
.code-block {
  align-items: stretch;
  display: flex;
  flex: 1 1 20em;
  flex-direction: column;
  max-height: tools.size('rhythm') * 12;
  max-width: 100%;
  overflow: hidden;
}

scss 
.code-header {

  color: tools.color('code');
  flex: 1 1 auto;
  flex-wrap: wrap;

  code {
    padding: tools.size('shim');
  }
}
